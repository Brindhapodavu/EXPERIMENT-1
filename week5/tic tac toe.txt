b=[' ']*9
p='X'

def show():
    for i in range(0,9,3):
        print("|".join(b[i:i+3]))
        if i<6: print("-----")

def win(x):
    w=[(0,1,2),(3,4,5),(6,7,8),(0,3,6),(1,4,7),(2,5,8),(0,4,8),(2,4,6)]
    return any(b[a]==b[c]==b[d]==x for a,c,d in w)

while True:
    show()
    m=int(input(f"Player {p} move (0-8): "))
    if b[m]==' ':
        b[m]=p
        if win(p):
            show(); print(p,"wins"); break
        if ' ' not in b:
            show(); print("Draw"); break
        p='O' if p=='X' else 'X'
    else:
        print("Invalid move")
